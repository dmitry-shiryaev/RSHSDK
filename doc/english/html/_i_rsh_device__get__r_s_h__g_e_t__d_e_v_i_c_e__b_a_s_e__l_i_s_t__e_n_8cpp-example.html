IRshDevice_Get_RSH_GET_DEVICE_BASE_LIST_EN.cpp

Example of use IRshDevice::Get() method to get base adress list.

IRshDevice* device;
...
//get device interface
...
RSH_BUFFER_U32 baseList;
//unnecessary call, default constructor will make buffer empty anyway
baseList.SetSize(0);
//get base adress list
U32 res = device->Get(RSH_GET_DEVICE_BASE_LIST, &baseList);
if(res != RSH_API_SUCCESS)
{
//process error
}
if(baseList.Size() == 0)
{
//base list is empty, no device connected or something else is wrong
}
else
{
//Connect to first device in list
RshDeviceKey connectionKey = RshDeviceKey(baseList[0]);
U32 res = device->Connect(&connectionKey);
}
