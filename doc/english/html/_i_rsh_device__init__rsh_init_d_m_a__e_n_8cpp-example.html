IRshDevice_Init_RshInitDMA_EN.cpp

Example of IRshDevice initialization. RshInitDMA stucture is used.

IRshDevice* device;
//initialization structure
RshInitDMA params;
params.startType = RshInitDMA::Program; //start from internal timer
params.frequency = 10000.0; //sampling rate (per channel)
params.dmaMode = RshInitDMA::Single; //single start
params.bufferSize = 2048; //buffer size (per channel)
//channels setup
//set channel 0 as active
params.channels[0]SetUsed();
//Alternativly one can manipulate bits in channel's control word directly
//params.channels[0].control |= RshChannel::Used;
U32 res = device->Init(&params);
if(res != RSH_API_SUCESS)
{
//process error
}
