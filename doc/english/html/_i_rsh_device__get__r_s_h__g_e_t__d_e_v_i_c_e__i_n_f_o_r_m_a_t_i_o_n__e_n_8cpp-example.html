IRshDevice_Get_RSH_GET_DEVICE_INFORMATION_EN.cpp

Example of use IRshDevice::Get() method to get various information about device and abstraction library.

IRshDevice* device;
...
//get interface, connect, initialization
...
//get product ID of the device
RSH_U32 pid = 0;
U32 res = device->Get(RSH_GET_DEVICE_PID, &pid);
if(res != RSH_API_SUCCESS)
{
//process error
}
else
{
std::cout << "pid=0x" << std::hex << pid << std::endl;
}
//get vendor ID of the device
RSH_U32 vid = 0;
res = device->Get(RSH_GET_DEVICE_VID, &vid);
if(res != RSH_API_SUCCESS)
{
//process error
}
else
{
std::cout << "vid=0x" << std::hex << vid << std::endl;
}
//Get full name of abstraction library
RSH_S8P libName = 0;
res = device->Get(RSH_GET_DEVICE_NAME_VERBOSE, &libName);
if(res != RSH_API_SUCCESS)
{
//process error
}
else
{
std::cout << libName << std::endl;
}
