FrameFrequencyMode_EN.cpp

Sample of setup frame sampling mode using RshInitDMA init structure.

//initialization structure
RshInitDMA params;
//get maximum sampling rate
RSH_DOUBLE maxfreq;
U32 res = pADC->Get(RSH_GET_DEVICE_MAX_FREQUENCY, &maxfreq);
if(res != RSH_API_SUCESS)
{
//process error
}
params.startType = RshInitDMA::Program; //start programmly
params.frequency = 1000.0; //frame sampling rate
params.frequencyFrame = maxfreq; //sampling rate inside frame (set maximum available
params.dmaMode = RshInitDMA::Single; //single start
params.bufferSize = 256; //buffer size (per channel)
params.control.SetFrameMode(); //turn on frame frequency mode
//channel setup
params.channels[0].SetUsed();
params.channels[1].SetUsed();
params.channels[2].SetUsed();
res = pADC->Init(&params);
if(res != RSH_API_SUCESS)
{
//process error
}
