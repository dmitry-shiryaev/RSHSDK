RshError_EN.cpp

Example of usage RshError for obtaining error description based on error code.
Many functions in Rsh Api return combined code, containing two parts: low word ( 0xFFFF mask) is system error code, and high word ( 0xFFFF0000 mask) is one of the RSH_STATUS codes.

//print error message with code RSH_API_BUFFER_ISEMPTY
RshError::PrintError(RSH_API_BUFFER_ISEMPTY, RSH_LANGUAGE_RUSSIAN);
//get error message for system error code 38
std::wstring systemMessage;
U32 res = RshError::GetSystemErrorDescription(38, systemMessage);
if(res == RSH_API_SUCCESS)
{
//print it
std::wcout << L"System error with code 38:" << systemMessage << std::endl;
}
else
{
//error!
std::wcout << L"Couldn't format error message!" << std::endl;
}
//Example of processing return code of Rsh Api method
RSH_BUFFER_S16 buff;
buff.Allocate(64);
//file name is wrong deliberately
res = buff.WriteBufferToFile("/:/:/");
if(res != RSH_API_SUCCESS)
{
//print error description and system error description
RshError::PrintError(res, RSH_LANGUAGE_RUSSIAN, true);
}
//print debug message on the same code
RshError::PrintErrorDebug(res, "WriteBufferToFile");
//get error description (in english) in ANSI format
char buffc[512];
RshError::GetErrorDescription(res, buffc, 512);
