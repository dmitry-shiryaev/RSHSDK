RshInitDAC_EN.cpp

Sample of DAC control via RshInitDAC initialization structure.

IRshDevice* device;
...
//get interface, connect to device
...
//get DAC output range
RSH_DOUBLE range = 0;
if (device->Get(RSH_GET_DEVICE_OUTPUT_RANGE_VOLTS, &range) != RSH_API_SUCCESS)
{
//process error
}
// init structure used to control DACs
RshInitDAC par;
par.id = 0; //write to first (zero index) DAC
par.voltage = range/2.0; //set output voltage as half maximum
//Initialization (data will be send to DAC output here)
if(device->Init(&par) != RSH_API_SUCCESS)
{
//process error
}
//and once more
par.voltage = -3.7;
device->Init(&par);
