IRshDevice_Get_RSH_GET_DEVICE_BASE_LIST_EXT_EN.cpp

Example of use IRshDevice::Get() method to get list of structures with information about devices connected.

IRshDevice* device;
...
//get device interface
...
RSH_BUFFER_DEVICE_BASE_INFO baseInfoBuffer;
baseInfoBuffer.SetSize(0);
//get list of structures with extended information about devices
U32 res = device->Get(RSH_GET_DEVICE_BASE_LIST_EXT, &baseInfoBuffer);
if(res != RSH_API_SUCCESS)
{
//process error
}
if(baseInfoBuffer.Size() == 0)
{
//list is empty, no device connected or something other is wrong
}
else
{
//output information from buffer
std::cout << baseInfoBuffer << std::endl;
}
