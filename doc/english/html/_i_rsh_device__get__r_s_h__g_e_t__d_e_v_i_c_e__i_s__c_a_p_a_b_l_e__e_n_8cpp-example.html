IRshDevice_Get_RSH_GET_DEVICE_IS_CAPABLE_EN.cpp

Example of use IRshDevice::Get() method to check device's feature support.

IRshDevice* device;
...
//get interface
...
//check if device library support RshInitMemory structure
RSH_U32 caps = RSH_CAPS_SOFT_INIT_MEMORY;
if(device->Get(RSH_GET_DEVICE_IS_CAPABLE, &caps) == RSH_API_SUCCESS)
{
//we can init device using RshInitMemory structure
}
// one more
if(device->Get(RSH_GET_DEVICE_IS_CAPABLE, &(caps=RSH_CAPS_DEVICE_SIZE_LIST)) == RSH_API_SUCCESS)
{
//device has fixed size list, we can obtain it
RSH_BUFFER_U32 sizeList;
device->Get(RSH_GET_DEVICE_SIZE_LIST, &sizeList);
}
