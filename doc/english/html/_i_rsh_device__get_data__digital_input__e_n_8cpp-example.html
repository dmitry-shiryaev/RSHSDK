IRshDevice_GetData_DigitalInput_EN.cpp

Example of using additional flags to get digital port input data within analog data samples

IRshDevice* device;
...
//get interface, connect to device
...
//get data bits of the device
RSH_U32 dataBits = 0;
if (device->Get(RSH_GET_DEVICE_DATA_BITS, &dataBits) != RSH_API_SUCCESS)
{
//process error
}
//create data mask and ditial port mask (for 16 bit data)
U16 dataMask = (U16)0xFFFF << (16 - dataBits);
U16 portMask = (U16)0xFFFF ^ dataMask;
//total digital lines available in data will be equal to 16 - dataBits
//allocate memory
RSH_BUFFER_S16 buffer = RSH_BUFFER_S16(1024);
...
//init, start acquisition, wait for buffer ready event
...
//get data buffer
U32 res = device->GetData(&buffer, RSH_DATA_MODE_CONTAIN_DIGITAL_INPUT);
if(res != RSH_API_SUCCESS)
{
//process error
}
else
{
//we can get access to ditial port data in every sample
for(U32 i=0;i<buffer.Size();++i)
{
//check if we have signal on zero bit of the ditial port
if(buffer.data[i] & 0x1)
{
//do something with data
}
}
}
