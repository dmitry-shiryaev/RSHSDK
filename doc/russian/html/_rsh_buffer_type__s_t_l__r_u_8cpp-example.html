RshBufferType_STL_RU.cpp

Использование алгоритмов STL с RshBufferType.

#include <iostream>
#include <iomanip>
#include <algorithm>
#include <RshBufferType.h>
//Функтор-генератор "степени двойки"
struct c_poweroftwo {
int current;
c_poweroftwo() {current=0;}
int operator()() {return 1 << current++;}
};
//целочисленный буфер RshBufferType с размером m_psize = 10
RSH_BUFFER_U16 buffer(10);
//Заполним буфер случайными числами
//Вызов данной функции также меняет актуальный размер буфера , m_size будет равным 10 после вызова
//buffer.end() указывает на конец актуальной части буфера (m_size+1), а не физической (m_psize)+1.
RshFillBufferWithRandomNumbers(buffer);
std::cout << "Buffer filled with random numbers:" << std::endl;
std::cout << buffer << std::endl;
//поиск первого элемента в буфере, значение которого больше 10000, ипользуя метод STL
//вместо RSH_BUFFER_U16::value_type можно использовать просто U16
std::cout << std::endl << "Find value greater than 10000:" << std::endl;
RSH_BUFFER_U16::iterator it = std::find_if(buffer.begin(), buffer.end(), std::bind2nd(std::greater<RSH_BUFFER_U16::value_type>(), 10000));
if( it != buffer.end())
std::cout << "Found value greater than 10000 at position " << it - buffer.begin() << ", value = " << *it << std::endl;
//заполнение буфера степенями двойки
c_poweroftwo PowerOfTwo;
std::generate(buffer.begin(), buffer.end(), PowerOfTwo);
std::cout << std::endl << "Buffer filled with powers of two:" << std::endl;
std::cout << buffer << std::endl;
//замена всех элементов, значение которых меньше 80, на 80
std::cout << std::endl << "Replace all values less than 80 with 80" << std::endl;
std::replace_if(buffer.begin(), buffer.end(), std::bind2nd(std::less<U16>(), 80), 80);
std::cout << buffer << std::endl;
//удаление дубликатов из буфера
std::cout << std::endl << "Remove duplicates from buffer" << std::endl;
it = std::unique(buffer.begin(), buffer.end() );
//Изменение размера (std::unique не меняет размер буфера, просто переставляет ненужные элементы в конец)
size_t newSize = it - buffer.begin();
buffer.SetSize(newSize);
std::cout << buffer << std::endl;
