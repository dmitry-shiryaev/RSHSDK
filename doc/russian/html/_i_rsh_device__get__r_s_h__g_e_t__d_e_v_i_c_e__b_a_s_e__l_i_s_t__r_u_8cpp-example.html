IRshDevice_Get_RSH_GET_DEVICE_BASE_LIST_RU.cpp

Пример использования метода IRshDevice::Get() для получения списка допустимых базовых адресов.

IRshDevice* device;
...
//получение интерфейса устройства
...
RSH_BUFFER_U32 baseList;
//необязательный вызов, конструктор по-умолчанию делает буфер пустым
baseList.SetSize(0);
//получим список допустимых базовых адресов
U32 res = device->Get(RSH_GET_DEVICE_BASE_LIST, &baseList);
if(res != RSH_API_SUCCESS)
{
//обработка ошибки
}
if(baseList.Size() == 0)
{
//список адресов пуст, устройство не подключено или какая-то другая проблема
}
else
{
//подключаемся к первому устройству
RshDeviceKey connectionKey = RshDeviceKey(baseList[0]);
U32 res = device->Connect(&connectionKey);
}
