IRshDevice_Get_RSH_GET_DEVICE_BASE_LIST_EXT_RU.cpp

Пример использования метода IRshDevice::Get() для получения массива структур с расширенной информацией об устройствах.

IRshDevice* device;
...
//получение интерфейса устройства
...
RSH_BUFFER_DEVICE_BASE_INFO baseInfoBuffer;
baseInfoBuffer.SetSize(0);
//получим массив структур с расширенной информацией о подключенных устройствах
U32 res = device->Get(RSH_GET_DEVICE_BASE_LIST_EXT, &baseInfoBuffer);
if(res != RSH_API_SUCCESS)
{
//обработка ошибки
}
if(baseInfoBuffer.Size() == 0)
{
//список пуст, устройство не подключено или какая-то другая проблема
}
else
{
//вывод информации из буфера
std::cout << baseInfoBuffer << std::endl;
}
