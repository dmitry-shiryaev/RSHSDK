IRshDevice_Get_RSH_GET_BUFFER_READY_RU.cpp

Пример использования метода IRshDevice::Get() с кодом RSH_GET_BUFFER_READY для проверки готовности буфера данных.

IRshDevice* device;
...
//получение интерфейса, подключение, инициализация
...
RSH_U32 ready = 0;
//запускаем сбор
device->Start();
do{
U32 res = device->Get(RSH_GET_BUFFER_READY, &ready);
if(res != RSH_API_SUCCESS)
{
//обработка ошибки и выход из цикла
}
}while (ready == 0)
//получение данных
device->GetData(...);
