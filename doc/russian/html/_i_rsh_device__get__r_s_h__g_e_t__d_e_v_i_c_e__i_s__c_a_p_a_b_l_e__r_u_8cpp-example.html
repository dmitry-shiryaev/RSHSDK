IRshDevice_Get_RSH_GET_DEVICE_IS_CAPABLE_RU.cpp

Пример использования метода IRshDevice::Get() для определения поддержки устройством различных возможностей.

IRshDevice* device;
...
//получение интерфейса
...
//проверка поддержки библиотекой структуры инициализации RshInitMemory
RSH_U32 caps = RSH_CAPS_SOFT_INIT_MEMORY;
if(device->Get(RSH_GET_DEVICE_IS_CAPABLE, &caps) == RSH_API_SUCCESS)
{
//можно делать инициализацию используя структуру RshInitMemory
}
// и еще
if(device->Get(RSH_GET_DEVICE_IS_CAPABLE, &(caps=RSH_CAPS_DEVICE_SIZE_LIST)) == RSH_API_SUCCESS)
{
//устройство имеет список фиксированных размеров буфера, получим его
RSH_BUFFER_U32 sizeList;
device->Get(RSH_GET_DEVICE_SIZE_LIST, &sizeList);
}
