RshError_RU.cpp

Пример использования класса RshError для получения описания ошибок по их коду.
Многие функции Rsh Api в качестве результата возвращают комбинированнный код, состоящий из двух частей: в младшем слове (маска 0xFFFF ) содержится код системной ошибки, а в старшем (маска 0xFFFF0000) - один из кодов RSH_STATUS.

//вывод сообщения об ошибке с кодом RSH_API_BUFFER_ISEMPTY
RshError::PrintError(RSH_API_BUFFER_ISEMPTY, RSH_LANGUAGE_RUSSIAN);
//получим описание системной ошибки с кодом 38
std::wstring systemMessage;
U32 res = RshError::GetSystemErrorDescription(38, systemMessage);
if(res == RSH_API_SUCCESS)
{
//напечатаем его
std::wcout << L"System error with code 38:" << systemMessage << std::endl;
}
else
{
//ошибка!
std::wcout << L"Couldn't format error message!" << std::endl;
}
//пример обработки возвращаемого кода фукнции из Rsh Api
RSH_BUFFER_S16 buff;
buff.Allocate(64);
//вызов функции с заведомо неправильным именем файла
res = buff.WriteBufferToFile("/:/:/");
if(res != RSH_API_SUCCESS)
{
//печать сообщения об ошибке (с дополнительным выводом информации о системной ошибке)
RshError::PrintError(res, RSH_LANGUAGE_RUSSIAN, true);
}
//печать отладочного сообщения для того же кода
RshError::PrintErrorDebug(res, "WriteBufferToFile");
//получим сообщение об ошибке (на английском) в ANSI формате
char buffc[512];
RshError::GetErrorDescription(res, buffc, 512);
