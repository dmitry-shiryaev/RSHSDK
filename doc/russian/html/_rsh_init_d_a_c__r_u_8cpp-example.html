RshInitDAC_RU.cpp

Пример управления ЦАПом с помощью структуры инициализации RshInitDAC

IRshDevice* device;
...
//получение интерфейса, подключение к устройству
...
//получим выходной диапазон ЦАПа
RSH_DOUBLE range = 0;
if (device->Get(RSH_GET_DEVICE_OUTPUT_RANGE_VOLTS, &range) != RSH_API_SUCCESS)
{
//обработка ошибки
}
//структура инициализации для управления ЦАПами
RshInitDAC par;
par.id = 0; //запись в нулевой ЦАП
par.voltage = range/2.0; //половина диапазона
//Инициализация (выдача напряжения на ЦАП)
if(device->Init(&par) != RSH_API_SUCCESS)
{
//обработка ошибки
}
//еще один раз
par.voltage = -3.7;
device->Init(&par);
