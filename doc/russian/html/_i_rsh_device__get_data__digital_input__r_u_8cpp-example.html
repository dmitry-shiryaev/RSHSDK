IRshDevice_GetData_DigitalInput_RU.cpp

Пример получения данных цифрового порта вместе с данными аналоговых каналов.

IRshDevice* device;
...
//получение интерфейса, подключение к устройству
...
//получим разрядность АЦП
RSH_U32 dataBits = 0;
if (device->Get(RSH_GET_DEVICE_DATA_BITS, &dataBits) != RSH_API_SUCCESS)
{
//обработка ошибки
}
//построение маски аналоговых данных и маски данных цифрового порта (для 16 битного слова данных)
U16 dataMask = (U16)0xFFFF << (16 - dataBits);
U16 portMask = (U16)0xFFFF ^ dataMask;
//общее количество линий цифрового порта( битов), которые будут передаваться с данными будет равно 16 - dataBits
//выделим память под данные
RSH_BUFFER_S16 buffer = RSH_BUFFER_S16(1024);
...
//инициализация, запуск преобразования, ожидание готовности буфера
...
//получаем данные
U32 res = device->GetData(&buffer, RSH_DATA_MODE_CONTAIN_DIGITAL_INPUT);
if(res != RSH_API_SUCCESS)
{
//обработка ошибки
}
else
{
//можно получить доступ к данным цифрового порта в каждом отсчетче (слове данных)
for(U32 i=0;i<buffer.Size();++i)
{
//проверим, установлен ли нулевой бит (первая линия цифрового порта)
if(buffer.data[i] & 0x1)
{
//делаем что-нибудь с данными
}
}
}
